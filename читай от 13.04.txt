javascript
// 1. ПРОВЕРКА КОДА СПАСЕНИЯ (ОРБИТА)
if (rawVal === "142.0458.13.70") { 
    console.log("ВЕКТОР ПРИНЯТ: ОРБИТА СТАБИЛИЗИРОВАНА");

    // ВАЖНО: Мы НЕ удаляем s9_emergency_flag здесь.
    // Мы только помечаем, что первая часть квеста выполнена.
    localStorage.setItem('s9_orbit_stabilized', 'active');

    const label = document.getElementById('statusLabel');
    if (label) {
        const genUI = document.getElementById('gen-ui');
        const readUI = document.getElementById('read-ui');
        const isReadMode = document.getElementById('mode-switch').checked;

        // Показываем статус, скрываем инпут
        if (genUI) genUI.classList.add('hidden');
        if (readUI) readUI.classList.remove('hidden');
        
        // НОВЫЙ ТЕКСТ: предупреждаем о реакторе
        label.innerText = "ОРБИТА СТАБИЛЬНА. ВНИМАНИЕ: ОШИБКА ЯДРА РЕАКТОРА!";
        label.style.color = "#ffaa00"; // Оранжевый цвет тревоги
        label.style.textShadow = "0 0 15px #ffaa00";
        
        // Через 4 секунды возвращаем интерфейс в норму, но станция остается КРАСНОЙ
        setTimeout(() => {
            label.innerText = "соедините точки";
            label.style.color = "";
            label.style.textShadow = "";

            if (!isReadMode && genUI && readUI) {
                genUI.classList.remove('hidden');
                readUI.classList.add('hidden');
            }
            
            // Мы НЕ делаем reload(), чтобы пользователь мог уйти на карту сам
        }, 4000);
    }

    e.target.value = ""; // Очищаем поле ввода
    return;
}
